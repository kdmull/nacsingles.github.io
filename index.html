Welcome to Nitro Pickleball

This app needs a free Supabase database to sync scores across all players. Setup takes about 3 minutes:

  1. Go to supabase.com → New Project
  2. Open SQL Editor and run the query below
  3. Go to Settings → API and copy your URL + anon key
  4. Click Connect Database and paste them in
CREATE TABLE pb_league (
  key TEXT PRIMARY KEY,
  value JSONB NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE pb_league ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read"   ON pb_league FOR SELECT USING (true);
CREATE POLICY "public write"  ON pb_league FOR INSERT WITH CHECK (true);
CREATE POLICY "public update" ON pb_league FOR UPDATE USING (true);
